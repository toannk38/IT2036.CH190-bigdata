\begin{frame}{Aggregation \& API Service}
    \begin{block}{Aggregation Service}
        \begin{itemize}
            \item Kết hợp điểm số từ AI/ML và LLM engines
            \item Tính toán weighted score: \texttt{0.6 * technical + 0.4 * sentiment}
            \item Tạo alerts dựa trên ngưỡng điểm số
            \item Lưu trữ kết quả vào MongoDB
        \end{itemize}
    \end{block>
    
    \begin{block}{REST API Service}
        \begin{itemize}
            \item FastAPI framework
            \item Endpoints: \texttt{/stocks}, \texttt{/analysis}, \texttt{/alerts}
            \item JSON response format
            \item Authentication \& rate limiting
        \end{itemize}
    \end{block>
