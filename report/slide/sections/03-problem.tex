\begin{frame}{Thách thức và cơ hội}
    \begin{block}{Thách thức}
        \begin{itemize}
            \item Dữ liệu phân tán từ nhiều nguồn
            \item Tin tức tài chính bằng tiếng Việt
            \item Cần xử lý real-time với độ trễ thấp
            \item Tích hợp nhiều loại phân tích (kỹ thuật + sentiment)
        \end{itemize}
    \end{block}
    
    \begin{block}{Cơ hội}
        \begin{itemize}
            \item Ứng dụng AI/ML trong tài chính
            \item Microservices architecture
            \item Cloud-native deployment
            \item Open source ecosystem
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Mục tiêu dự án}
    \begin{enumerate}
        \item \textbf{Thu thập dữ liệu:} Tự động thu thập giá cổ phiếu và tin tức
        \item \textbf{Xử lý real-time:} Sử dụng Apache Kafka cho message streaming
        \item \textbf{Phân tích AI:} Kết hợp phân tích kỹ thuật và sentiment analysis
        \item \textbf{Cảnh báo thông minh:} Tạo điểm số tổng hợp và alerts
        \item \textbf{API service:} Cung cấp REST API cho ứng dụng client
    \end{enumerate}
