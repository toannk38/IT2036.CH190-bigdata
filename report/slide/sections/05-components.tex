\section{Các thành phần hệ thống}

\begin{frame}{Data Collection Layer}
    \begin{block}{Price Collector (5-minute interval)}
        \begin{itemize}
            \item Thu thập dữ liệu giá từ vnstock API
            \item Xử lý 30 mã cổ phiếu VN30
            \item Lưu trữ: Open, High, Low, Close, Volume
            \item Gửi message qua Kafka topic: \texttt{stock\_prices\_raw}
        \end{itemize}
    \end{block}
    
    \begin{block}{News Collector (30-minute interval)}
        \begin{itemize}
            \item Thu thập tin tức từ các nguồn tài chính
            \item Lọc tin tức liên quan đến VN30
            \item Xử lý văn bản tiếng Việt
            \item Gửi message qua Kafka topic: \texttt{stock\_news\_raw}
        \end{itemize}
    \end{block}
