\section{Giới thiệu}

\begin{frame}{Giới thiệu dự án}
    \begin{itemize}
        \item \textbf{Tên dự án:} Vietnam Stock AI Backend
        \item \textbf{Mục tiêu:} Xây dựng hệ thống phân tích chứng khoán Việt Nam tự động
        \item \textbf{Công nghệ:} Microservices, AI/ML, Real-time Processing
        \item \textbf{Phạm vi:} Thu thập, xử lý và phân tích dữ liệu chứng khoán VN30
    \end{itemize}
    
    \vspace{1em}
    \begin{block}{Đặc điểm nổi bật}
        \begin{itemize}
            \item Thu thập dữ liệu real-time từ vnstock API
            \item Phân tích sentiment từ tin tức tài chính
            \item Tính toán điểm số tổng hợp và cảnh báo
            \item Kiến trúc microservices với Docker
        \end{itemize}
    \end{block}
\end{frame}
